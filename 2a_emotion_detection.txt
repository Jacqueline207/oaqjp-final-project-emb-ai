import requests
# here write the function to run emotion detection 
#using the appropriatemotion Detection function. name the function emotion_detector
def emotion_detector(text_to_analyze):
    #assume text to be analysed is passed to the function
    #as an argument and stored in the variable text to analyze
    #the value being returned must be the text attribute of the response
    #object as recived from the emotion detection function
    #NEED URL, HEADERS, IMPORT JSON, and the RESPONSE
    url = "https://sn-watson-emotion.labs.skills.network/v1/watson.runtime.nlp.v1/NlpService/EmotionPredict"
    headers = {
        "grpc-metadata-mm-model-id": "emotion_aggregated-workflow_lang_en_stock"
        }
    input_json = {
        "raw_document": {
             "text": text_to_analyze
            }
    }
    #requests.post to send to watson emotion api
    response = requests.post(url, json=input_json, headers=headers)
    #text
    return response.text
