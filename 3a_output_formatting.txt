import requests
import json
# here write the function to run emotion detection 
#using the appropriatemotion Detection function. name the function emotion_detector
def emotion_detector(text_to_analyze):
    #assume text to be analysed is passed to the function
    #as an argument and stored in the variable text to analyze
    #the value being returned must be the text attribute of the response
    #object as recived from the emotion detection function
    #NEED URL, HEADERS, IMPORT JSON, and the RESPONSE
    url = "https://sn-watson-emotion.labs.skills.network/v1/watson.runtime.nlp.v1/NlpService/EmotionPredict"
    headers = {
        "grpc-metadata-mm-model-id": "emotion_aggregated-workflow_lang_en_stock"
        }
    input_json = {
        "raw_document": {
             "text": text_to_analyze
            }
    }
    #requests.post to send to watson emotion api
    response = requests.post(url, json=input_json, headers=headers)
    #text

    #first step is to convert the respnse text into a dictionary, using the json functions. I can use loads
    response_dictionary = json.loads(response.text)

    #becuase there are multiple emotions such as emotion prediction and emotion mention just get the first?
    emotions = response_dictionary["emotionPredictions"][0]["emotion"]
    #get individual scores from the dictionary
    anger = emotions["anger"]
    disgust = emotions["disgust"]
    fear = emotions["fear"]
    joy = emotions["joy"]
    sadness = emotions["sadness"]

    #next i'll store the scores again in a new dictionary

    emotion_scores_dict = {
        "anger": anger,
        "disgust" : disgust,
        "fear": fear,
        "joy" : joy,
        "sadness" : sadness
    }
    #to find the dominant is as simple of just getting the max of all of them
    dominant_emotion = max(emotion_scores_dict, key=emotion_scores_dict.get)# max uses ()

    #then i'll create a new dictionary that includes the dominant emotion as well and return that instesad
    result = {
        "anger": anger,
        "disgust" : disgust,
        "fear": fear,
        "joy" : joy,
        "sadness" : sadness,
        "dominant_emotion" : dominant_emotion
    }

    return result
